Circular Linked List

Structure

structure Node:
    value
    next

Insert at Head (Circular)
function InsertCircular(head, newNode):
    if head == NULL:
        newNode.next = newNode
        return newNode

    temp = head
    while temp.next != head:
        temp = temp.next

    temp.next = newNode
    newNode.next = head
    head = newNode
    return head

Insert at Tail (Circular)

function InsertCircularTail(head, newNode):
    if head == NULL:
        newNode.next = newNode
        return newNode
    
    temp = head
    while temp.next != head:
        temp = temp.next
    
    temp.next = newNode
    newNode.next = head
    return head

function DeleteCircular(head, key):
    if head == NULL:
        return NULL

    curr = head
    prev = NULL

    // Special case: delete head
    if head.value == key:
        temp = head
        while temp.next != head:
            temp = temp.next
        
        if head.next == head:
            free(head)
            return NULL
        
        temp.next = head.next
        newHead = head.next
        free(head)
        return newHead

    // General case
    curr = head.next
    prev = head

    while curr != head:
        if curr.value == key:
            prev.next = curr.next
            free(curr)
            return head

        prev = curr
        curr = curr.next

    return head
