function Dijkstra(Graph, source):

    // 1. Initialization
    for each vertex v in Graph:
        dist[v] = INFINITY
        prev[v] = NULL

    dist[source] = 0

    // Min-priority queue storing (distance, vertex)
    PQ = new PriorityQueue()
    PQ.enqueue(source, 0)

    // Optional: visited set (avoids reprocessing)
    visited = empty set

    // 2. Process vertices
    while PQ is not empty:
        (u, d) = PQ.extractMin()   // u = vertex with smallest d

        if u in visited:
            continue
        add u to visited

        // 3. Relax edges
        for each neighbor v of u:
            alt = dist[u] + weight(u, v)

            if alt < dist[v]:
                dist[v] = alt
                prev[v] = u
                PQ.enqueue(v, dist[v])

    return dist, prev
