function InsertionSort(arr):

    n = length(arr)

    for i from 1 to n - 1:
        key = arr[i]
        j = i - 1

        // Shift elements greater than key to the right
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j = j - 1

        // Place key in correct position
        arr[j + 1] = key
